DELIMITER //

CREATE PROCEDURE return_book (
  IN p_rollno INT,
  IN p_bookname VARCHAR(200)
)
BEGIN
    DECLARE v_date_of_issue DATE;
    DECLARE v_days INT DEFAULT 0;
    DECLARE v_fine DECIMAL(10,2) DEFAULT 0.00;
    DECLARE v_status CHAR(1);

    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        SELECT 'An error occurred. Please check inputs or try again.' AS ErrorMessage;
    END;

    -- Try fetching the issue info
    SELECT DateofIssue, Status
    INTO v_date_of_issue, v_status
    FROM Borrower
    WHERE Rollno = p_rollno AND NameofBook = p_bookname
    LIMIT 1;

    -- If no matching record found
    IF v_date_of_issue IS NULL THEN
        SELECT CONCAT('No issued record found for Rollno=', p_rollno, ' and Book=', p_bookname) AS Message;
    ELSE
        -- If already returned
        IF v_status = 'R' THEN
            SELECT 'Book has already been returned.' AS Message;
        ELSE
            -- Calculate fine
            SET v_days = DATEDIFF(CURDATE(), v_date_of_issue);

            IF v_days > 30 THEN
                SET v_fine = (15 * 5) + ((v_days - 30) * 50);
            ELSEIF v_days > 15 THEN
                SET v_fine = (v_days - 15) * 5;
            ELSE
                SET v_fine = 0;
            END IF;

            -- Update borrower status
            UPDATE Borrower
            SET Status = 'R'
            WHERE Rollno = p_rollno AND NameofBook = p_bookname;

            -- Insert fine if applicable
            IF v_fine > 0 THEN
                INSERT INTO Fine (Roll_no, FineDate, Amt)
                VALUES (p_rollno, CURDATE(), v_fine);
            END IF;

            -- Final message
            SELECT 'Book returned successfully.' AS Message, v_days AS Days_Borrowed, v_fine AS Fine;
        END IF;
    END IF;
END;
//

DELIMITER ;

CREATE PROCEDURE calculate_area()
BEGIN
    -- Declare local variables
    DECLARE radius DECIMAL(5,2);
    DECLARE area DECIMAL(10,2);
    DECLARE i INT DEFAULT 5;         -- Starting radius value

    -- WHILE loop runs from radius = 5 to 9
    WHILE i <= 9 DO
        SET radius = i;                             -- Assign radius
        SET area = 3.14159 * radius * radius;       -- Area = πr²
        INSERT INTO areas(radius, area)             -- Store result
        VALUES (radius, area);
        SET i = i + 1;                              -- Increment radius
    END WHILE;
END //
DELIMITER ;

CREATE DATABASE IF NOT EXISTS ass5;

CREATE TABLE Stud_Marks (
    name VARCHAR(100),
    total_marks INT
);

CREATE TABLE Result (
    Roll INT AUTO_INCREMENT PRIMARY KEY,
    Name VARCHAR(100),
    Class VARCHAR(50)
);

DELIMITER //

CREATE FUNCTION Grade_Categorization (marks INT) 
RETURNS VARCHAR(50)
DETERMINISTIC
BEGIN
    DECLARE grade VARCHAR(50);

    -- User-defined exception for invalid marks
    DECLARE CONTINUE HANDLER FOR SQLEXCEPTION
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Invalid Marks Value';

    -- Validate marks range
    IF marks < 0 OR marks > 1500 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Marks must be between 0 and 1500';
    END IF;

    -- Categorize grade based on marks
    IF marks >= 990 AND marks <= 1500 THEN
        SET grade = 'Distinction';
    ELSEIF marks >= 900 AND marks <= 989 THEN
        SET grade = 'First Class';
    ELSEIF marks >= 825 AND marks <= 899 THEN
        SET grade = 'Higher Second Class';
    ELSE
        SET grade = 'Not Categorized';
    END IF;

    RETURN grade;
END //

DELIMITER ;

DELIMITER //

CREATE PROCEDURE proc_Grade()
BEGIN
    DECLARE done INT DEFAULT 0;
    DECLARE v_name VARCHAR(100);
    DECLARE v_total_marks INT;
    DECLARE v_grade VARCHAR(50);

    -- Declare cursor to iterate through student marks table
    DECLARE student_cursor CURSOR FOR 
        SELECT name, total_marks FROM Stud_Marks;

    -- Handler for end of records
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;

    -- Handler for SQL exceptions
    DECLARE CONTINUE HANDLER FOR SQLEXCEPTION
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Error occurred during procedure execution';

    -- Open cursor
    OPEN student_cursor;

    -- Loop through each record
    read_loop: LOOP
        FETCH student_cursor INTO v_name, v_total_marks;
        IF done THEN
            LEAVE read_loop;
        END IF;

        -- Call the function for grade categorization
        BEGIN
            DECLARE CONTINUE HANDLER FOR SQLEXCEPTION
                SET v_grade = 'Error in Grade Categorization';
            SET v_grade = Grade_Categorization(v_total_marks);
        END;

        -- Insert results into table
        INSERT INTO Result (Name, Class) VALUES (v_name, v_grade);
    END LOOP;

    -- Close cursor after processing
    CLOSE student_cursor;

    -- Commit changes
    COMMIT;
END //

DELIMITER ;

CREATE PROCEDURE MergeRollCallData()
BEGIN
    -- Declare local variables
    DECLARE done INT DEFAULT 0;
    DECLARE n_id INT;
    DECLARE n_name VARCHAR(255);
    DECLARE n_date DATE;

    -- Declare cursor to read all rows from N_RollCall
    DECLARE cur CURSOR FOR
        SELECT id, name, roll_call_date FROM N_RollCall;

    -- Handler to detect end of data
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;

    -- Open the cursor
    OPEN cur;

    -- Loop through all rows from new table
    read_loop: LOOP
        FETCH cur INTO n_id, n_name, n_date;

        -- Exit when all rows are processed
        IF done THEN
            LEAVE read_loop;
        END IF;

        -- Check if record exists in old table before inserting
        IF NOT EXISTS (
            SELECT 1 
            FROM O_RollCall 
            WHERE id = n_id 
              AND name = n_name 
              AND roll_call_date = n_date
        ) THEN
            INSERT INTO O_RollCall (id, name, roll_call_date)
            VALUES (n_id, n_name, n_date);
        END IF;
    END LOOP;

    -- Close cursor after completion
    CLOSE cur;
END $$

DELIMITER ;

CREATE DATABASE IF NOT EXISTS trigger_audit_db;
USE trigger_audit_db;
-- Create main table to store library books
CREATE TABLE library (
    library_id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL
);

-- Create audit table to store history of updates and deletions
CREATE TABLE library_audit (
    audit_id INT AUTO_INCREMENT PRIMARY KEY,
    action_type ENUM('update', 'delete') NOT NULL,
    library_id INT NOT NULL,
    old_title VARCHAR(255),
    action_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Trigger: Before Update (logs the old title before any modification)
DELIMITER //
CREATE TRIGGER before_library_update
BEFORE UPDATE ON library
FOR EACH ROW
BEGIN
    -- Insert old record data into audit table before updating
    INSERT INTO library_audit (action_type, library_id, old_title)
    VALUES ('update', OLD.library_id, OLD.title);
END //
DELIMITER ;

-- Trigger: Before Delete (logs the old title before deletion)
DELIMITER //
CREATE TRIGGER before_library_delete
BEFORE DELETE ON library
FOR EACH ROW
BEGIN
    -- Insert old record data into audit table before deletion
    INSERT INTO library_audit (action_type, library_id, old_title)
    VALUES ('delete', OLD.library_id, OLD.title);
END //
DELIMITER ;

-- Insert a sample record
INSERT INTO library (title)
VALUES ('to kill a mockingbird');

-- Update record to fire BEFORE UPDATE trigger
UPDATE library
SET title = '1984'
WHERE title = 'to kill a mockingbird';

-- Delete record to fire BEFORE DELETE trigger
DELETE FROM library
WHERE title = '1984';

-- Display log entries from the audit table
SELECT * FROM library_audit;

        cursor.close()
        conn.close()
        
        return redirect(url_for('index'))
