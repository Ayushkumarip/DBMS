Library Fine Calculation

CREATE TABLE Borrower (
    Roll_no INT,
    Name VARCHAR2(100),
    DateofIssue DATE,
    NameofBook VARCHAR2(200),
    Status CHAR(1) -- 'I' for Issued, 'R' for Returned
);

CREATE TABLE Fine (
    Roll_no INT,
    "Date" DATE,
    Amt NUMBER
);

-- Insert some sample data to test
INSERT INTO Borrower VALUES (101, 'Student A', SYSDATE - 20, 'Database Systems', 'I'); -- 20 days
INSERT INTO Borrower VALUES (102, 'Student B', SYSDATE - 45, 'Java Programming', 'I'); -- 45 days
INSERT INTO Borrower VALUES (103, 'Student C', SYSDATE - 10, 'Data Structures', 'I');   -- 10 days

DECLARE
    -- Variables to hold user input
    v_roll_no     Borrower.Roll_no%TYPE := &input_roll_no;
    v_book_name   Borrower.NameofBook%TYPE := '&input_book_name';
    
    -- Variables for calculations
    v_date_of_issue Borrower.DateofIssue%TYPE;
    v_days_late     NUMBER;
    v_fine_amount   Fine.Amt%TYPE := 0;

BEGIN
    -- 1. Get the issue date for the specific book and student
    SELECT DateofIssue INTO v_date_of_issue
    FROM Borrower
    WHERE Roll_no = v_roll_no
      AND NameofBook = v_book_name
      AND Status = 'I'; -- Only process books that are currently 'Issued'

    -- 2. Calculate the number of days (TRUNC removes time component)
    v_days_late := TRUNC(SYSDATE) - TRUNC(v_date_of_issue);

    -- 3. Check fine logic
    IF v_days_late > 30 THEN
        -- Rs 5 for first 30 days + Rs 50 for days after 30
        v_fine_amount := (30 * 5) + ((v_days_late - 30) * 50);
        
    ELSIF v_days_late >= 15 AND v_days_late <= 30 THEN
        -- Rs 5 per day
        v_fine_amount := v_days_late * 5;
        
    END IF; -- If v_days_late < 15, fine remains 0
    
    -- 4. Update the book status from 'I' (Issued) to 'R' (Returned)
    UPDATE Borrower
    SET Status = 'R'
    WHERE Roll_no = v_roll_no
      AND NameofBook = v_book_name
      AND Status = 'I';

    -- 5. If a fine is due, insert it into the Fine table
    IF v_fine_amount > 0 THEN
        INSERT INTO Fine (Roll_no, "Date", Amt)
        VALUES (v_roll_no, SYSDATE, v_fine_amount);
        
        DBMS_OUTPUT.PUT_LINE('Book returned. Days late: ' || v_days_late || '. Fine of Rs. ' || v_fine_amount || ' applied.');
    ELSE
        DBMS_OUTPUT.PUT_LINE('Book returned. Days late: ' || v_days_late || '. No fine.');
    END IF;
    
    COMMIT; -- Make the changes permanent

EXCEPTION
    -- 6. Exception Handling
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Error: No active record found for Roll No: ' || v_roll_no || ' with book: ' || v_book_name);
        ROLLBACK;
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('An unexpected error occurred: ' || SQLERRM);
        ROLLBACK;
END;
/

Stored Procedure (proc_Grade)

CREATE OR REPLACE PROCEDURE proc_Grade (
    p_roll_no IN Stud_Marks.Roll_no%TYPE
)
AS
    v_name        Stud_Marks.Name%TYPE;
    v_marks       Stud_Marks.total_marks%TYPE;
    v_class       Result.Class%TYPE;
BEGIN
    -- 1. Get the student's details from the Stud_Marks table
    SELECT Name, total_marks
    INTO v_name, v_marks
    FROM Stud_Marks
    WHERE Roll_no = p_roll_no;

    -- 2. Determine the class based on the marks
    IF v_marks >= 990 AND v_marks <= 1500 THEN
        v_class := 'Distinction';
    ELSIF v_marks >= 900 AND v_marks <= 989 THEN
        v_class := 'First Class';
    ELSIF v_marks >= 825 AND v_marks <= 899 THEN
        v_class := 'Higher Second Class';
    ELSE
        v_class := 'Other'; -- For marks below 825
    END IF;

    -- 3. Insert the categorized data into the Result table
    INSERT INTO Result (Roll, Name, Class)
    VALUES (p_roll_no, v_name, v_class);
    
    DBMS_OUTPUT.PUT_LINE('Successfully processed Roll No: ' || p_roll_no);

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Error: No student found with Roll No: ' || p_roll_no);
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('An unexpected error occurred: ' || SQLERRM);
END proc_Grade;

— PL/SQL Block with Parameterized Cursor

DECLARE
    -- 1. Define the Parameterized Cursor.
    -- It takes a roll number as a parameter and checks
    -- if it exists in the O_RollCall table.
    CURSOR c_check_old (p_roll_no IN O_RollCall.Roll_no%TYPE) IS
        SELECT 1 
        FROM O_RollCall 
        WHERE Roll_no = p_roll_no;
        
    -- Variable to hold the result of the cursor fetch
    v_exists NUMBER;

BEGIN
    -- 2. Use a Cursor FOR Loop to iterate through all records in the new table.
    -- 'rec' is an implicitly defined record variable.
    FOR rec IN (SELECT Roll_no, Name FROM N_RollCall) LOOP
    
        -- 3. Open the parameterized cursor, passing the current roll number
        OPEN c_check_old(rec.Roll_no);
        
        -- 4. Fetch the result into the v_exists variable
        FETCH c_check_old INTO v_exists;
        
        -- 5. Check the cursor's %NOTFOUND attribute.
        -- If TRUE, it means the fetch found no matching record.
        IF c_check_old%NOTFOUND THEN
            -- Data does NOT exist, so we MERGE (INSERT) it.
            INSERT INTO O_RollCall (Roll_no, Name)
            VALUES (rec.Roll_no, rec.Name);
            
            DBMS_OUTPUT.PUT_LINE('MERGED: Roll No ' || rec.Roll_no || ' (' || rec.Name || ')');
        ELSE
            -- Data already exists, so we SKIP it.
            DBMS_OUTPUT.PUT_LINE('SKIPPED: Roll No ' || rec.Roll_no || ' (' || rec.Name || ') already exists.');
        END IF;
        
        -- 6. Close the parameterized cursor before the next loop iteration
        CLOSE c_check_old;
        
    END LOOP;
    
    COMMIT; -- Save the changes made by the INSERT statements

EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('An error occurred: ' || SQLERRM);
        ROLLBACK;
END;
/

-- 1. Create the main Library table
CREATE TABLE Library (
    book_id INT PRIMARY KEY,
    title VARCHAR2(200),
    author VARCHAR2(100),
    status VARCHAR2(50) -- e.g., 'Available', 'On Loan'
);

-- 2. Create the table to store the audit trail
CREATE TABLE Library_Audit (
    old_book_id INT,
    old_title VARCHAR2(200),
    old_author VARCHAR2(100),
    old_status VARCHAR2(50),
    action_type VARCHAR2(10), -- 'UPDATE' or 'DELETE'
    action_timestamp TIMESTAMP,  -- When the change happened
    action_user VARCHAR2(100) -- Who made the change
);

— The Database Trigger

CREATE OR REPLACE TRIGGER trg_Library_Audit
-- Fire AFTER a DELETE or UPDATE operation
AFTER DELETE OR UPDATE ON Library
-- This makes it a ROW-LEVEL trigger
FOR EACH ROW
DECLARE
    v_action VARCHAR2(10);
BEGIN
    -- 1. Determine which action (DELETE or UPDATE) fired the trigger
    IF DELETING THEN
        v_action := 'DELETE';
    ELSIF UPDATING THEN
        v_action := 'UPDATE';
    END IF;
    
    -- 2. Insert the OLD values into the audit table.
    -- The :OLD record contains the data as it was *before*
    -- the DELETE or UPDATE statement.
    INSERT INTO Library_Audit (
        old_book_id,
        old_title,
        old_author,
        old_status,
        action_type,
        action_timestamp,
        action_user
    )
    VALUES (
        :OLD.book_id,
        :OLD.title,
        :OLD.author,
        :OLD.status,
        v_action,
        SYSTIMESTAMP,
        USER
    );

EXCEPTION
    WHEN OTHERS THEN
        -- Handle any unexpected errors during the trigger execution
        DBMS_OUTPUT.PUT_LINE('Error in trg_Library_Audit: ' || SQLERRM);
END trg_Library_Audit;
/

MongoDB - Aggregation and Indexing
———————————————————————————————————————
Aggregation

The Aggregation Pipeline is used to process data through multiple stages.
Example: Find the average marks for each subject, and sort them from highest average to lowest.
db.students.aggregate([
  // Stage 1: Group documents by the 'subject' field
  {
    $group: {
      _id: "$subject", // The field to group by
      averageMarks: { $avg: "$marks" }, // Calculate the average of 'marks' for each group
      studentCount: { $sum: 1 } // Count the number of students in each group
    }
  },
  
  // Stage 2: Sort the resulting groups
  {
    $sort: { averageMarks: -1 } // Sort by averageMarks in descending order (-1)
  }
]);

Indexing

Indexes speed up read queries by providing a fast lookup path.
Create a Single-Field Index: This makes queries that filter by city much faster.
// 1 = ascending order
db.students.createIndex({ city: 1 });
Create a Compound Index: This is useful for queries that filter by subject and sort by marks.
// 1 = ascending (for subject), -1 = descending (for marks)
db.students.createIndex({ subject: 1, marks: -1 });

db.students.getIndexes();

MongoDB - MapReduce Operations

——————————————————————————————————————————————

Step 1: The map function This function is run on every document. It emits a key (the subject) and a value (the marks).
var mapFunction = function() {
    emit(this.subject, this.marks);
};

Step 2: The reduce function This function takes all the values emitted for a single key (e.g., "Math", [85, 78, 95]) and "reduces" them to a single result.
var reduceFunction = function(keySubject, valuesMarks) {
    // valuesMarks is an array of all marks for that keySubject
    return Array.avg(valuesMarks);
};

Step 3: Run the MapReduce operation This command runs the functions and stores the final results in a new collection named subject_averages.
db.students.mapReduce(
    mapFunction,
    reduceFunction,
    { out: "subject_averages" } // Output the results to this collection
);

// To see the results, you then query the new collection:
db.subject_averages.find();
